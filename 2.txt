        public Zanrovi_i_Glumci()
        {
            InitializeComponent();
            //relative paths of base linking
            string putanja = Environment.CurrentDirectory;
            string[] putanjaBaze = putanja.Split(new string[] { "bin" }, StringSplitOptions.None);
            AppDomain.CurrentDomain.SetData("DataDirectory", putanjaBaze[0]);
            konekcija = new OleDbConnection(@"Provider=Microsoft.ACE.OLEDB.12.0;Data Source=|DataDirectory|\DVDTesanovic.accdb");

            konekcija.Open();
            komanda = new OleDbCommand("Select NazivZanra from ZANR order by NazivZanra ASC", konekcija);
            rider = komanda.ExecuteReader();
            while (rider.Read())
            {
                comboBox1.Items.Add(rider[0]);
            }
            konekcija.Close();
           

        }


        private void button1_Click(object sender, EventArgs e)
        {
            listView1.Items.Clear();
            konekcija.Open();
            string q1 = "SELECT FILM.NAZIV, TIP_ULOGE.Tip, ULOGA.ImeLika ";
            string q2 = "FROM ZANR INNER JOIN(TIP_ULOGE INNER JOIN (Glumac INNER JOIN ((Producent INNER JOIN (FILM INNER JOIN Producirao ON FILM.FilmID=Producirao.FilmID) ON Producent.ProducentID=Producirao.ProducentID) INNER JOIN ULOGA ON FILM.FilmID=ULOGA.FilmID) ON Glumac.GlumacID=Uloga.GlumacID) ON TIP_ULOGE.TipUlogeID=Uloga.TipUlogeID) ON ZANR.ZanrID=FILM.ZanrID ";
            string q3 = "where ZANR.NazivZanra=@nz AND Glumac.Ime=@ime AND Glumac.Prezime=@pr ORDER BY FILM.Trajanje DESC";
            komanda = new OleDbCommand(q1 + q2 + q3, konekcija);
            komanda.Parameters.AddWithValue("@nz", comboBox1.Text);
            komanda.Parameters.AddWithValue("@ime", textBox1.Text);
            komanda.Parameters.AddWithValue("@pr", textBox2.Text);
            try
            {
                rider = komanda.ExecuteReader();
            }
            catch (Exception ex)
            {
                MessageBox.Show(ex.Message);
            }
            while (true)
            {
                try
                {
                    rider.Read();
                    string[] stavka = { rider[0].ToString(), rider[1].ToString(), rider[2].ToString() };
                    ListViewItem lv = new ListViewItem(stavka);
                    listView1.Items.Add(lv);
                }
                catch (Exception ex)
                {
                    break;
                }

            }
            konekcija.Close();
           
        }
